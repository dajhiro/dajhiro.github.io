```ts
@UseGuards(SessionAuthGuard)
async createPost(
  @Body() input: UploadRequestDto,
  @Req() req: Request
) {
  const userId = req.user!.id;
  this.logger.log(
    `Received request to create from URL: ${input.youtubeUrl} (user: ${userId})`
  )
  
  this.uploadService
    .createPostFromUrl(input, userId)
    .catch((error) => {
      this.logger.error('Background post creation failed', error);
    });
}
```
